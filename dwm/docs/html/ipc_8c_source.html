43static int ipc_create_socket(const char *filename)
44{
45 char *normal_filename;
46 char *parent;
47 const size_t addr_size = sizeof(struct sockaddr_un);
48 const int sock_type = SOCK_STREAM | SOCK_NONBLOCK | SOCK_CLOEXEC;
49
50 normalizepath(filename, &normal_filename);
51
52 // In case socket file exists
53 unlink(normal_filename);
54
55 // For portability clear the addr structure, since some implementations have
56 // nonstandard fields in the structure
57 memset(&sockaddr, 0, addr_size);
58
59 parentdir(normal_filename, &parent);
60 // Create parent directories
61 mkdirp(parent);
62 free(parent);
63
64 sockaddr.sun_family = AF_LOCAL;
65 strcpy(sockaddr.sun_path, normal_filename);
66 free(normal_filename);
67
68 sock_fd = socket(AF_LOCAL, sock_type, 0);
69 if (sock_fd == -1) {
70 fputs("Failed to create socket\n", stderr);
71 return -1;
72 }
73
74 DEBUG("Created socket at %s\n", sockaddr.sun_path);
75
76 if (bind(sock_fd, (const struct sockaddr *)&sockaddr, addr_size) == -1) {
77 fputs("Failed to bind socket\n", stderr);
78 return -1;
79 }
80
81 DEBUG("Socket binded\n");
82
83 if (listen(sock_fd, IPC_SOCKET_BACKLOG) < 0) {
84 fputs("Failed to listen for connections on socket\n", stderr);
85 return -1;
86 }
87
88 DEBUG("Now listening for connections on socket\n");
89
90 return sock_fd;
91}

101static int ipc_recv_message(int fd, uint8_t *msg_type, uint32_t *reply_size, uint8_t **reply)
102{
103 uint32_t read_bytes = 0;
104 const int32_t to_read = sizeof(jdwm_ipc_header_t);
105 char header[to_read];
106 char *walk = header;
107
108 // Try to read header
109 while (read_bytes < to_read) {
110 const ssize_t n = read(fd, header + read_bytes, to_read - read_bytes);
111
112 if (n == 0) {
113 if (read_bytes == 0) {
114 fprintf(stderr, "Unexpectedly reached EOF while reading header.");
115 fprintf(stderr, "Read %" PRIu32 " bytes, expected %" PRIu32 " total bytes.\n",
116 read_bytes, to_read);
117 return -2;
118 } else {
119 fprintf(stderr, "Unexpectedly reached EOF while reading header.");
120 fprintf(stderr, "Read %" PRIu32 " bytes, expected %" PRIu32 " total bytes.\n",
121 read_bytes, to_read);
122 return -3;
123 }
124 } else if (n == -1) {
125 // errno will still be set
126 return -1;
127 }
128
129 read_bytes += n;
130 }
131
132 // Check if magic string in header matches
133 if (memcmp(walk, IPC_MAGIC, IPC_MAGIC_LEN) != 0) {
134 fprintf(stderr, "Invalid magic string. Got '%.*s', expected '%s'\n", IPC_MAGIC_LEN, walk, IPC_MAGIC);
135 return -3;
136 }
137
138 walk += IPC_MAGIC_LEN;
139
140 // Extract reply size
141 memcpy(reply_size, walk, sizeof(uint32_t));
142 walk += sizeof(uint32_t);
143
144 if (*reply_size > MAX_MESSAGE_SIZE) {
145 fprintf(stderr, "Message too long: %" PRIu32 " bytes. ", *reply_size);
146 fprintf(stderr, "Maximum message size is: %d\n", MAX_MESSAGE_SIZE);
147 return -4;
148 }
149
150 // Extract message type
151 memcpy(msg_type, walk, sizeof(uint8_t));
152 walk += sizeof(uint8_t);
153
154 if (*reply_size > 0)
155 (*reply) = malloc(*reply_size);
156 else
157 return 0;
158
159 read_bytes = 0;
160 while (read_bytes < *reply_size) {
161 const ssize_t n = read(fd, *reply + read_bytes, *reply_size - read_bytes);
162
163 if (n == 0) {
164 fprintf(stderr, "Unexpectedly reached EOF while reading payload.");
165 fprintf(stderr, "Read %" PRIu32 " bytes, expected %" PRIu32 " bytes.\n", read_bytes,
166 *reply_size);
167 free(*reply);
168 return -2;
169 } else if (n == -1) {
170 // TODO: Should we return and wait for another epoll event?
171 // This would require saving the partial read in some way.
172 if (errno == EINTR || errno == EAGAIN || errno == EWOULDBLOCK) continue;
173
174 free(*reply);
175 return -1;
176 }
177
178 read_bytes += n;
179 }
180
181 return 0;
182}

192static ssize_t ipc_write_message(int fd, const void *buf, size_t count)
193{
194 size_t written = 0;
195
196 while (written < count) {
197 const ssize_t n = write(fd, (uint8_t *)buf + written, count - written);
198
199 if (n == -1) {
200 if (errno == EAGAIN || errno == EWOULDBLOCK)
201 return written;
202 else if (errno == EINTR)
203 continue;
204 else
205 return n;
206 }
207
208 written += n;
209 DEBUG("Wrote %zu/%zu to client at fd %d\n", written, count, fd);
210 }
211
212 return written;
213}

306static int ipc_parse_run_command(char *msg, IPCParsedCommand *parsed_command)
307{
308 char error_buffer[1000];
309 yajl_val parent = yajl_tree_parse(msg, error_buffer, 1000);
310
311 if (parent == NULL) {
312 fputs("Failed to parse command from client\n", stderr);
313 fprintf(stderr, "%s\n", error_buffer);
314 fprintf(stderr, "Tried to parse: %s\n", msg);
315 return -1;
316 }
317
318 // Format:
319 // {
320 // "command": "<command name>"
321 // "args": [ "arg1", "arg2", ... ]
322 // }
323 const char *command_path[] = { "command", 0 };
324 yajl_val command_val = yajl_tree_get(parent, command_path, yajl_t_string);
325
326 if (command_val == NULL) {
327 fputs("No command key found in client message\n", stderr);
328 yajl_tree_free(parent);
329 return -1;
330 }
331
332 const char *command_name = YAJL_GET_STRING(command_val);
333 size_t command_name_len = strlen(command_name);
334 parsed_command->name = (char *)malloc((command_name_len + 1) * sizeof(char));
335 strcpy(parsed_command->name, command_name);
336
337 DEBUG("Received command: %s\n", parsed_command->name);
338
339 const char *args_path[] = { "args", 0 };
340 yajl_val args_val = yajl_tree_get(parent, args_path, yajl_t_array);
341
342 if (args_val == NULL) {
343 fputs("No args key found in client message\n", stderr);
344 yajl_tree_free(parent);
345 return -1;
346 }
347
348 unsigned int *argc = &parsed_command->argc;
349 Arg **args = &parsed_command->args;
350 ArgType **arg_types = &parsed_command->arg_types;
351
352 *argc = args_val->u.array.len;
353
354 // If no arguments are specified, make a dummy argument to pass to the
355 // function. This is just the way dwm's void(Arg*) functions are setup.
356 if (*argc == 0) {
357 *args = (Arg *)malloc(sizeof(Arg));
358 *arg_types = (ArgType *)malloc(sizeof(ArgType));
359 (*arg_types)[0] = ARG_TYPE_NONE;
360 (*args)[0].i = 0;
361 (*argc)++;
362 } else if (*argc > 0) {
363 *args = (Arg *)calloc(*argc, sizeof(Arg));
364 *arg_types = (ArgType *)malloc(*argc * sizeof(ArgType));
365
366 for (int i = 0; i < *argc; i++) {
367 yajl_val arg_val = args_val->u.array.values[i];
368
369 if (YAJL_IS_NUMBER(arg_val)) {
370 if (YAJL_IS_INTEGER(arg_val)) {
371 // Any values below 0 must be a signed int
372 if (YAJL_GET_INTEGER(arg_val) < 0) {
373 (*args)[i].i = YAJL_GET_INTEGER(arg_val);
374 (*arg_types)[i] = ARG_TYPE_SINT;
375 DEBUG("i=%ld\n", (*args)[i].i);
376 // Any values above 0 should be an unsigned int
377 } else if (YAJL_GET_INTEGER(arg_val) >= 0) {
378 (*args)[i].ui = YAJL_GET_INTEGER(arg_val);
379 (*arg_types)[i] = ARG_TYPE_UINT;
380 DEBUG("ui=%ld\n", (*args)[i].i);
381 }
382 // If the number is not an integer, it must be a float
383 } else {
384 (*args)[i].f = (float)YAJL_GET_DOUBLE(arg_val);
385 (*arg_types)[i] = ARG_TYPE_FLOAT;
386 DEBUG("f=%f\n", (*args)[i].f);
387 // If argument is not a number, it must be a string
388 }
389 } else if (YAJL_IS_STRING(arg_val)) {
390 char *arg_s = YAJL_GET_STRING(arg_val);
391 size_t arg_s_size = (strlen(arg_s) + 1) * sizeof(char);
392 (*args)[i].v = (char *)malloc(arg_s_size);
393 (*arg_types)[i] = ARG_TYPE_STR;
394 strcpy((char *)(*args)[i].v, arg_s);
395 }
396 }
397 }
398
399 yajl_tree_free(parent);
400
401 return 0;
402}

425static int ipc_validate_run_command(IPCParsedCommand *parsed, const IPCCommand actual)
426{
427 if (actual.argc != parsed->argc) return -1;
428
429 for (int i = 0; i < parsed->argc; i++) {
430 ArgType ptype = parsed->arg_types[i];
431 ArgType atype = actual.arg_types[i];
432
433 if (ptype != atype) {
434 if (ptype == ARG_TYPE_UINT && atype == ARG_TYPE_PTR)
435 // If this argument is supposed to be a void pointer, cast it
436 parsed->args[i].v = (void *)parsed->args[i].ui;
437 else if (ptype == ARG_TYPE_UINT && atype == ARG_TYPE_SINT)
438 // If this argument is supposed to be a signed int, cast it
439 parsed->args[i].i = parsed->args[i].ui;
440 else
441 return -2;
442 }
443 }
444
445 return 0;
446}

481{

487 fprintf(stderr, "%s\n", error_buffer);

489 }
490
491 // Format:
492 // {
493 // "event": "<event name>"
494 // "action": "<subscribe|unsubscribe>"
495 // }
496 const char *event_path[] = { "event", 0 };
497 yajl_val event_val = yajl_tree_get(parent, event_path, yajl_t_string);

505 DEBUG("Received event: %s\n", event_str);

508
509 const char *action_path[] = { "action", 0 };
510 yajl_val action_val = yajl_tree_get(parent, action_path, yajl_t_string);
511
512 if (action_val == NULL) {
513 fputs("No 'action' key found in client message\n", stderr);
514 return -1;
515 }
516
517 const char *action = YAJL_GET_STRING(action_val);
518
519 if (strcmp(action, "subscribe") == 0)
520 *subscribe = IPC_ACTION_SUBSCRIBE;
521 else if (strcmp(action, "unsubscribe") == 0)
522 *subscribe = IPC_ACTION_UNSUBSCRIBE;
523 else {
524 fputs("Invalid action specified for subscription\n", stderr);
525 return -1;
526 }
527
528 yajl_tree_free(parent);
529
530 return 0;
531}

584static int ipc_run_command(IPCClient *ipc_client, char *msg)
585{
586 IPCParsedCommand parsed_command;
587 IPCCommand ipc_command;
588
589 // Initialize struct
590 memset(&parsed_command, 0, sizeof(IPCParsedCommand));
591
592 if (ipc_parse_run_command(msg, &parsed_command) < 0) {
593 ipc_prepare_reply_failure(ipc_client, IPC_TYPE_RUN_COMMAND, "Failed to parse run command");
594 return -1;
595 }
596
597 if (ipc_get_ipc_command(parsed_command.name, &ipc_command) < 0) {
598 ipc_prepare_reply_failure(ipc_client, IPC_TYPE_RUN_COMMAND, "Command %s not found",
599 parsed_command.name);
600 ipc_free_parsed_command_members(&parsed_command);
601 return -1;
602 }
603
604 int res = ipc_validate_run_command(&parsed_command, ipc_command);
605 if (res < 0) {
606 if (res == -1)
607 ipc_prepare_reply_failure(ipc_client, IPC_TYPE_RUN_COMMAND,
608 "%u arguments provided, %u expected", parsed_command.argc,
609 ipc_command.argc);
610 else if (res == -2)
611 ipc_prepare_reply_failure(ipc_client, IPC_TYPE_RUN_COMMAND, "Type mismatch");
612 ipc_free_parsed_command_members(&parsed_command);
613 return -1;
614 }
615
616 if (parsed_command.argc == 1)
617 ipc_command.func.single_param(parsed_command.args);
618 else if (parsed_command.argc > 1)
619 ipc_command.func.array_param(parsed_command.args, parsed_command.argc);
620
621 DEBUG("Called function for command %s\n", parsed_command.name);
622
623 ipc_free_parsed_command_members(&parsed_command);
624
625 ipc_prepare_reply_success(ipc_client, IPC_TYPE_RUN_COMMAND);
626 return 0;
627}

736int ipc_init(const char *socket_path, const int p_epoll_fd, IPCCommand commands[], const int commands_len)
737{
738 // Initialize struct to 0
739 memset(&sock_epoll_event, 0, sizeof(sock_epoll_event));
740
741 int socket_fd = ipc_create_socket(socket_path);
742 if (socket_fd < 0) return -1;
743
744 ipc_commands = commands;
745 ipc_commands_len = commands_len;
746
747 epoll_fd = p_epoll_fd;
748
749 // Wake up to incoming connection requests
750 sock_epoll_event.data.fd = socket_fd;
751 sock_epoll_event.events = EPOLLIN;
752 if (epoll_ctl(epoll_fd, EPOLL_CTL_ADD, socket_fd, &sock_epoll_event)) {
753 fputs("Failed to add sock file descriptor to epoll", stderr);
754 return -1;
755 }
756
757 return socket_fd;
758}

760void ipc_cleanup()
761{
762 IPCClient *c = ipc_clients;
763 // Free clients and their buffers
764 while (c) {
765 ipc_drop_client(c);
766 c = ipc_clients;
767 }
768
769 // Stop waking up for socket events
770 epoll_ctl(epoll_fd, EPOLL_CTL_DEL, sock_fd, &sock_epoll_event);
771
772 // Uninitialize all static variables
773 epoll_fd = -1;
774 sock_fd = -1;
775 ipc_commands = NULL;
776 ipc_commands_len = 0;
777 memset(&sock_epoll_event, 0, sizeof(struct epoll_event));
778 memset(&sockaddr, 0, sizeof(struct sockaddr_un));
779
780 // Delete socket
781 unlink(sockaddr.sun_path);
782
783 shutdown(sock_fd, SHUT_RDWR);
784 close(sock_fd);
785}

802int ipc_accept_client()
803{
804 int fd = -1;
805
806 struct sockaddr_un client_addr;
807 socklen_t len = 0;
808
809 // For portability clear the addr structure, since some implementations
810 // have nonstandard fields in the structure
811 memset(&client_addr, 0, sizeof(struct sockaddr_un));
812
813 fd = accept(sock_fd, (struct sockaddr *)&client_addr, &len);
814 if (fd < 0 && errno != EINTR) {
815 fputs("Failed to accept IPC connection from client", stderr);
816 return -1;
817 }
818
819 if (fcntl(fd, F_SETFD, FD_CLOEXEC) < 0) {
820 shutdown(fd, SHUT_RDWR);
821 close(fd);
822 fputs("Failed to set flags on new client fd", stderr);
823 }
824
825 IPCClient *nc = ipc_client_new(fd);
826 if (nc == NULL) return -1;
827
828 // Wake up to messages from this client
829 nc->event.data.fd = fd;
830 nc->event.events = EPOLLIN | EPOLLHUP;
831 epoll_ctl(epoll_fd, EPOLL_CTL_ADD, fd, &nc->event);
832
833 ipc_list_add_client(&ipc_clients, nc);
834
835 DEBUG("%s%d\n", "New client at fd: ", fd);
836
837 return fd;
838}

840int ipc_drop_client(IPCClient *c)
841{
842 int fd = c->fd;
843 shutdown(fd, SHUT_RDWR);
844 int res = close(fd);
845
846 if (res == 0) {
847 struct epoll_event ev;
848
849 // Stop waking up to messages from this client
850 epoll_ctl(epoll_fd, EPOLL_CTL_DEL, fd, &ev);
851 ipc_list_remove_client(&ipc_clients, c);
852
853 free(c->buffer);
854 free(c);
855
856 DEBUG("Successfully removed client on fd %d\n", fd);
857 } else if (res < 0 && res != EINTR) {
858 fprintf(stderr, "Failed to close fd %d\n", fd);
859 }
860
861 return res;
862}

864int ipc_read_client(IPCClient *c, IPCMessageType *msg_type, uint32_t *msg_size, char **msg)
865{
866 int fd = c->fd;
867 int ret = ipc_recv_message(fd, (uint8_t *)msg_type, msg_size, (uint8_t **)msg);
868
869 if (ret < 0) {
870 // This will happen if these errors occur while reading header
871 if (ret == -1 && (errno == EINTR || errno == EAGAIN || errno == EWOULDBLOCK)) return -2;
872
873 fprintf(stderr, "Error reading message: dropping client at fd %d\n", fd);
874 ipc_drop_client(c);
875
876 return -1;
877 }
878
879 // Make sure receive message is null terminated to avoid parsing issues
880 if (*msg_size > 0) {
881 size_t len = *msg_size;
882 nullterminate(msg, &len);
883 *msg_size = len;
884 }
885
886 DEBUG("[fd %d] ", fd);
887 if (*msg_size > 0)
888 DEBUG("Received message: '%.*s' ", *msg_size, *msg);
889 else
890 DEBUG("Received empty message ");
891 DEBUG("Message type: %" PRIu8 " ", (uint8_t)*msg_type);
892 DEBUG("Message size: %" PRIu32 "\n", *msg_size);
893
894 return 0;
895}
896
897ssize_t ipc_write_client(IPCClient *c)
898{
899 const ssize_t n = ipc_write_message(c->fd, c->buffer, c->buffer_size);
900
901 if (n < 0) return n;
902
903 // TODO: Deal with client timeouts
904
905 if (n == c->buffer_size) {
906 c->buffer_size = 0;
907 free(c->buffer);
908 // No dangling pointers!
909 c->buffer = NULL;
910 // Stop waking up when client is ready to receive messages
911 if (c->event.events & EPOLLOUT) {
912 c->event.events -= EPOLLOUT;
913 epoll_ctl(epoll_fd, EPOLL_CTL_MOD, c->fd, &c->event);
914 }
915 return n;
916 }
917
918 // Shift unwritten buffer to beginning of buffer and reallocate
919 c->buffer_size -= n;
920 memmove(c->buffer, c->buffer + n, c->buffer_size);
921 c->buffer = (char *)realloc(c->buffer, c->buffer_size);
922
923 return n;
924}

926void ipc_prepare_send_message(IPCClient *c, const IPCMessageType msg_type, const uint32_t msg_size, const char *msg)
927{
928 jdwm_ipc_header_t header = { .magic = IPC_MAGIC_ARR, .type = msg_type, .size = msg_size };
929
930 uint32_t header_size = sizeof(jdwm_ipc_header_t);
931 uint32_t packet_size = header_size + msg_size;
932
933 if (c->buffer == NULL)
934 c->buffer = (char *)malloc(c->buffer_size + packet_size);
935 else
936 c->buffer = (char *)realloc(c->buffer, c->buffer_size + packet_size);
937
938 // Copy header to end of client buffer
939 memcpy(c->buffer + c->buffer_size, &header, header_size);
940 c->buffer_size += header_size;
941
942 // Copy message to end of client buffer
943 memcpy(c->buffer + c->buffer_size, msg, msg_size);
944 c->buffer_size += msg_size;
945
946 // Wake up when client is ready to receive messages
947 c->event.events |= EPOLLOUT;
948 epoll_ctl(epoll_fd, EPOLL_CTL_MOD, c->fd, &c->event);
949}

1034void ipc_send_events(Monitor *mons, Monitor **lastselmon, Monitor *selmon)
1035{
1036 for (Monitor *m = mons; m; m = m->next) {
1037 unsigned int urg = 0, occ = 0, tagset = 0;
1038
1039 for (Client *c = m->clients; c; c = c->next) {
1040 occ |= c->tags;
1041
1042 if (c->isurgent) urg |= c->tags;
1043 }
1044 tagset = m->tagset[m->seltags];
1045
1046 TagState new_state = { .selected = tagset, .occupied = occ, .urgent = urg };
1047
1048 if (memcmp(&m->tagstate, &new_state, sizeof(TagState)) != 0) {
1049 ipc_tag_change_event(m->num, m->tagstate, new_state);
1050 m->tagstate = new_state;
1051 }
1052
1053 if (m->lastsel != m->sel) {
1054 ipc_client_focus_change_event(m->num, m->lastsel, m->sel);
1055 m->lastsel = m->sel;
1056 }
1057
1058 if (strcmp(m->ltsymbol, m->lastltsymbol) != 0 || m->lastlt != m->lt[m->sellt]) {
1059 ipc_layout_change_event(m->num, m->lastltsymbol, m->lastlt, m->ltsymbol, m->lt[m->sellt]);
1060 strcpy(m->lastltsymbol, m->ltsymbol);
1061 m->lastlt = m->lt[m->sellt];
1062 }
1063
1064 if (*lastselmon != selmon) {
1065 if (*lastselmon != NULL) ipc_monitor_focus_change_event((*lastselmon)->num, selmon->num);
1066 *lastselmon = selmon;

1073 .isfixed = sel->isfixed,

1091int ipc_handle_client_epoll_event(struct epoll_event *ev, Monitor *mons, Monitor **lastselmon, Monitor *selmon,
1092 const char *tags[], const int tags_len, const Layout *layouts, const int layouts_len)
1093{
1094 int fd = ev->data.fd;
1095 IPCClient *c = ipc_get_client(fd);
1096
1097 if (ev->events & EPOLLHUP) {
1098 DEBUG("EPOLLHUP received from client at fd %d\n", fd);
1099 ipc_drop_client(c);
1100 } else if (ev->events & EPOLLOUT) {
1101 DEBUG("Sending message to client at fd %d...\n", fd);
1102 if (c->buffer_size) ipc_write_client(c);
1103 } else if (ev->events & EPOLLIN) {
1104 IPCMessageType msg_type = 0;
1105 uint32_t msg_size = 0;
1106 char *msg = NULL;
1107
1108 DEBUG("Received message from fd %d\n", fd);
1109 if (ipc_read_client(c, &msg_type, &msg_size, &msg) < 0) return -1;
1110
1111 if (msg_type == IPC_TYPE_GET_MONITORS)
1112 ipc_get_monitors(c, mons, selmon);
1113 else if (msg_type == IPC_TYPE_GET_TAGS)
1114 ipc_get_tags(c, tags, tags_len);
1115 else if (msg_type == IPC_TYPE_GET_LAYOUTS)
1116 ipc_get_layouts(c, layouts, layouts_len);
1117 else if (msg_type == IPC_TYPE_RUN_COMMAND) {
1118 if (ipc_run_command(c, msg) < 0) return -1;
1119 ipc_send_events(mons, lastselmon, selmon);
1120 } else if (msg_type == IPC_TYPE_GET_DWM_CLIENT) {
1121 if (ipc_get_dwm_client(c, msg, mons) < 0) return -1;
1122 } else if (msg_type == IPC_TYPE_SUBSCRIBE) {
1123 if (ipc_subscribe(c, msg) < 0) return -1;
1124 } else {
1125 fprintf(stderr, "Invalid message type received from fd %d", fd);
1126 ipc_prepare_reply_failure(c, msg_type, "Invalid message type: %d", msg_type);
1127 }
1128 free(msg);
1129 } else {
1130 fprintf(stderr, "Epoll event returned %d from fd %d\n", ev->events, fd);
1131 return -1;
1132 }
1133
1134 return 0;
1135}
